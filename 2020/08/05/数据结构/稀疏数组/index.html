稀疏数组

基本介绍：

当一个数组大部分元素为0，或者为同一个值的数组时,可以用稀疏数组来保存

处理方式

  1. 记录数组一共几行几列，有多少个不同的值，把具体不同值的元元素的行列以及值，记录在一个小规模的数组中，从而缩小规模

代码实现

 public static void main(String[] args) &#123;
        //创建一个11 * 11 的二维数组
        //0表示没有棋子， 1表示黑子， 2表示白子
        int list[][] = new int[11][11];
        list[1][2] = 1;
        list[2][3] = 2;
        list[2][2] = 1;

        System.out.println("-------------原来数组---------------");
        for(int row[]: list)&#123;

                System.out.printf("%d\t",data);
           &#125;

       &#125;


        int[][] sparseArray = ArrayToSparseArray(list);
        System.out.println();
        System.out.println("-------------稀疏数组---------------");

  1. 数组转稀疏数组
    private static int[][] ArrayToSparseArray(int list[][])&#123;
     int value = 0;
    for(int i=0; i<list.length; i++)&#123;
        for(int j=0; j<list[0].length; j++)&#123;
            if(list[i][j] != 0)&#123;
                value++;
           &#125;
       &#125;
   &#125;
     int[][] sparseArray = new int[value+1][3];
     sparseArray[0][0] = list.length;
     sparseArray[0][1] = list[0].length;
     sparseArray[0][2] = value;


    int count = 0;
    for(int i=0; i<list.length; i++)&#123;
        for(int j=0; j<list[0].length; j++)&#123;
            if(list[i][j] != 0)&#123;
                count++;
                sparseArray[count][0]=i;
                sparseArray[count][1]=j;
                sparseArray[count][2]=list[i][j];
           &#125;
       &#125;
   &#125;
    return sparseArray;
