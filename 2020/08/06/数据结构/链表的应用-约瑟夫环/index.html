public class JosePhus &#123;
    IList<JosePhusNode> circle;   //约瑟夫环
    int size;

    public JosePhus(int[][] data)&#123;
        size = data.length;
        circle = new SingleList<>();
        for(int i=0; i<size; i++)&#123;
            circle.addNode(new JosePhusNode(data[i][0],data[i][1]));   // [no][pwd]
        &#125;
    &#125;

    //初始
    public JosePhus(int size)&#123;
        circle = new SingleList<>();
        this.size = size;
        int pwd;
        for(int i = 1; i<=size; i++)&#123;
            //为每人生成密码
            pwd = (int)(Math.random()*5)+1;
            circle.addNode(new JosePhusNode(i,pwd));
        &#125;
    &#125;


    //输出约瑟夫环
    public void printcircle()&#123;
        for(int i=0; i<size; i++)&#123;
            System.out.print("(" + circle.getNode(i).getNo() +"," + circle.getNode(i).getPwd() + ")");
        &#125;
    &#125;

    //出环
    public void startgame()&#123;
        int index = 0;
        JosePhusNode node = circle.getNode(0);
        for(int i=1; i<=size; i++)&#123;
            index = (--index + node.getPwd()) % circle.size();
            System.out.println("index:" + index + " size:" + circle.size());
            node = circle.remove(index);
            System.out.println("(" + node.getNo() + "," + node.getPwd() + ")出列");
